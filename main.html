Código de Algoritmo Genético

# Ejemplo de código Python (vos reemplazás luego)
import random

def fitness(x):
    return x*x

def genetic_algorithm():
    poblacion = [random.randint(0, 100) for _ in range(10)]
    return max(poblacion, key=fitness)

print(genetic_algorithm())
